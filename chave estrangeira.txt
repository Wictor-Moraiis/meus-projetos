create database chave_estrangeira;
use chave_estrangeira

CREATE TABLE aluno(
aluno_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
nome VARCHAR(30) NOT NULL);


CREATE TABLE cursos(
curso_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
nome VARCHAR(30) NOT NULL);

CREATE TABLE notas(
data DATE NOT NULL,
curso_id INT NOT NULL,
aluno_id INT NOT NULL,
nota DOUBLE NOT NULL,
PRIMARY KEY(data),
FOREIGN KEY (aluno_id) REFERENCES aluno(aluno_id) ON DELETE CASCADE ON UPDATE CASCADE,
FOREIGN KEY (curso_id) REFERENCES cursos(curso_id) ON DELETE CASCADE ON UPDATE CASCADE);
